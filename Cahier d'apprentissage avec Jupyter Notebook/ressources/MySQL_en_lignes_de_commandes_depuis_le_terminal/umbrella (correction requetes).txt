--Sylvie Durand devient Sylvie Dupont

--tous les employés du service Administratif
SELECT employe.nom, employe.prenom FROM employe
NATURAL JOIN service
WHERE service.libelle = 'Administratif';

--Tous les employés de Hollande arrivés à partir de 2011
SELECT employe.nom, employe.prenom FROM employe
JOIN site ON employe.id_site = site.id_site
WHERE employe.embauche >= 2011 AND site.nom = 'Hollande';

-- Tous les employés en vacances à partir du 01/01/2019
SELECT employe.nom, employe.prenom FROM employe
NATURAL JOIN vacances
WHERE vacances.fin >= '2019-01-01';
