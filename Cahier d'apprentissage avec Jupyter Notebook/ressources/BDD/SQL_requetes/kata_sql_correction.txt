//Trouver tous les arrêts du bus BB01.
SELECT bus.numero, arrets.nom FROM bus
JOIN lignes ON bus.id_ligne = lignes.id_ligne
JOIN arret_ligne ON lignes.id_ligne = arret_ligne.id_ligne
JOIN arrets ON arret_ligne.id_arret = arrets.id_arret
WHERE bus.numero = 'BB01';

//compte le nombre de bus
SELECT COUNT(*) AS nb_bus FROM bus;

//Compter le nombre d'arrêts de la ligne bleu.
SELECT COUNT(DISTINCT arrets.nom) AS arrets_bleu FROM arrets
JOIN arret_ligne ON arrets.id_arret = arret_ligne.id_arret
JOIN lignes ON arret_ligne.id_ligne = lignes.id_ligne
WHERE lignes.nom = 'Bleu';

//Trouver toutes les lignes qui passe par Morgana.
SELECT lignes.nom, arrets.nom FROM lignes
JOIN arret_ligne ON lignes.id_ligne = arret_ligne.id_ligne
JOIN arrets ON arret_ligne.id_arret = arrets.id_arret
WHERE arrets.nom = 'Morgana';
